173  def ImportGraphml(self, path):
174  nodes = {} # Dictionary of found nodes, key is unique index from GraphML
175  # Grab the root-element, and iterate over node-elements in graph-element
176  for xmlNode in ET.parse(path).getroot().find('{http://graphml.graphdrawing.org/xmlns}graph').findall('{http://graphml.graphdrawing.org/xmlns}node'):
177  # Read the unique index for this node
178  i = xmlNode.attrib['id']
179  # Geometry-element and label-element not found yet
180  xmlGeometry = None
181  xmlLabel = None
182  # Iterate over the data-elements in the node-element
183  for xmlData in xmlNode:
184  # Skip if this data-element do not contains any shape-element
185  xmlShape = xmlData.find('{http://www.yworks.com/xml/graphml}ShapeNode')
186  if xmlShape == None:
187  continue
188  # If a shape-element is found, grab the geometry-element and the label-element
189  xmlGeometry = xmlShape.find('{http://www.yworks.com/xml/graphml}Geometry')
190  xmlLabel = xmlShape.find('{http://www.yworks.com/xml/graphml}NodeLabel')
191  # If geometry-element or the label-element are not found, ignore this node
192  if xmlGeometry == None or xmlLabel == None:
193  print('[WARNING] Unable to decode node with id=\"%s\" which has been ignored' % (i))
194  # Read the name and the position of this node
195  name = xmlLabel.text
196  x = xmlGeometry.attrib['x']
197  y = xmlGeometry.attrib['y']
198  # Create the node and put it in the dictionary
199  nodes[i] = self.CreateNode(name = name, x = float(x), y = float(y))
200  # Grab the root-element, and iterate over edge-elements in graph-element
201  for xmlEdge in ET.parse(path).getroot().find('{http://graphml.graphdrawing.org/xmlns}graph').findall('{http://graphml.graphdrawing.org/xmlns}edge'):
202  # Read the unique id of this edge, of the source node and of the target node
203  iEdge = xmlEdge.attrib['id']
204  iSource = xmlEdge.attrib['source']
205  iTarget = xmlEdge.attrib['target']
206  # Find source and target node in the dictionary with their unique id
207  nodeSource = nodes[iSource]
208  nodeTarget = nodes[iTarget]
209  # Arrow-elements not found yet
210  xmlArrows = None
211  # Iterate over data-element in unknown-element in this edge-element to find any arrows-element
212  for xmlData in xmlEdge:
213  for xmlUnknown in xmlData:
214  xmlArrows = xmlUnknown.find('{http://www.yworks.com/xml/graphml}Arrows')
215  if xmlArrows == None:
216  continue
217  # If no arrows-element has been found, the edge will be bidirectional
218  if xmlArrows == None:
219  print('[WARNING] Unable to find direction of edge with id=\"%s\" which as been added as bidirectional between node with id=\"%s\" and node with id=\"%s\"' % (i, iSource, iTarget))
220  sourceToTarget = True
221  targetToSource = True
222  # If target-attribute of the arrows-element is 'none', consider the edge not oriented from source to target
223  sourceToTarget = not xmlArrows.attrib['target'] == 'none'
224  # If source-attribute of the arrows-element is 'none', consider the edge not oriented from target to source
225  targetToSource = not xmlArrows.attrib['source'] == 'none'
226  # If no orientation is found of two orientations are found, create a bidirectional edge and save it
227  if not sourceToTarget ^ targetToSource:
