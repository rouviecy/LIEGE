52  // List of created edges
53  vector<Edge*> resultat;
54  if(neighbourNode == NULL){
55  cout << "[WARNING] A null-pointer for neighbourNode as been provided to connect the node \"" << name << "\" ; this connection will be ignored" << endl;

77 void Node::DeleteEdgesToNode(Node* notNeighbourAnymore, bool unregisterReverseIncomingEdge){
78  // Iterate over outgoing edges
79  for(vector<Edge*>::iterator edge = edges.begin(); edge != edges.end();){
80  // Ignore edges with other target node
81  if((*edge)->GetToNode() != notNeighbourAnymore){
82  ++edge;
83  continue;
84  }
85  // Ask target node to unregister this edge from its incoming edges list
86  if(unregisterReverseIncomingEdge){
87  (*edge)->GetToNode()->UnregisterIncomingEdge(*edge);
88  }
89  // Edges are created in node context. They have to be deleted here.
90  delete *edge;
91  edge = edges.erase(edge);
92  }
93  // Iterate over incoming edges
94  for(vector<Edge*>::iterator edge = incomingEdges.begin(); edge != incomingEdges.end();){
95  // Ignore edges with other source node
96  if((*edge)->GetFromNode() != notNeighbourAnymore){
97  ++edge;
98  continue;
99  }
100  // Do not delete incoming edges here : they are created in the other node context. Let them delete these edges theirselves.
101  edge = incomingEdges.erase(edge);
102  }
103 }
